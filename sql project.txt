CREATE TABLE customerss (
customer_id NUMBER,
customer_name VARCHAR2(50)
);

CREATE TABLE storess (
store_id NUMBER,
store_name VARCHAR2(50)
);

CREATE TABLE productss (
product_id NUMBER,
category VARCHAR2(50)
);

CREATE TABLE saless (
sale_id NUMBER,
customer_id NUMBER,
store_id NUMBER,
product_id NUMBER,
sale_date DATE,
amount NUMBER,
quantity NUMBER,
discount NUMBER
);

-- Analyze time-based trends.
-- Questions:
-- 111.	Find daily sales.
SELECT 
    sale_date,
    SUM(amount)
FROM SALESS
GROUP BY sale_date;
-- 112.	Find monthly sales.
SELECT 
    TRUNC(sale_date,'MM'),
    SUM(amount)
FROM saless
GROUP BY TRUNC(sale_date,'MM');
-- 113.	Find weekend sales.
SELECT *
FROM saless
WHERE TO_CHAR(sale_date,'DY') IN ('SAT','SUN');
-- 114.	Find first sale date.
select min(sale_date) from saless 
-- 115.	Find last sale date.
select max(sale_date) from saless
-- 116.	Find sales in specific month.
SELECT *
FROM saless
WHERE TO_CHAR(sale_date,'MM') = '01';
-- 117.	Find sales growth month-wise.
SELECT 
    month,
    total_sales,
    total_sales - LAG(total_sales) OVER (ORDER BY month) growth
FROM (
    SELECT 
        TRUNC(sale_date,'MM') month,
        SUM(amount) total_sales
    FROM saless
    GROUP BY TRUNC(sale_date,'MM')
);
-- 118.	Find inactive customers (30 days).
SELECT customer_id
FROM saless
GROUP BY customer_id
HAVING MAX(sale_date) < SYSDATE - 30;
-- 119.	Find repeat customers.
SELECT customer_id
FROM saless
GROUP BY customer_id
HAVING COUNT(*) > 1;
-- 120.	Find sales trend.
SELECT 
    sale_date,
    SUM(amount) OVER (ORDER BY sale_date) AS trend
FROM saless;
